class Solution {
public:
    int numTrees(int n) {
        //TC=o(n^2) and SC=o(n)
        //formula of catalian no.
        // E T(i-1)*T(n-i-1);
        //making a array to store the catalian  no
        vector<int>arr(n+1);
        arr[0]=arr[1]=1;
        //filling the array
        for(int i=2;i<=n;i++){
            int sum=0;
            for(int j=0;j<i;j++)
                sum+=arr[j]*arr[i-j-1];
          arr[i]=sum;  
        }
        return arr[n];
    }
};
